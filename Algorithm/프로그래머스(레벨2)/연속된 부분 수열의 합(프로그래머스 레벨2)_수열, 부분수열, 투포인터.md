import java.util.*;

class Solution {
    public int[] solution(int[] sequence, int k) {
        ArrayList<int[]> list = new ArrayList<>();
        
        int start = 0;
        int end = 0;
        int sum = 0;
        
        while (start <= end && end < sequence.length) {
            sum += sequence[end];
            
            if (sum >= k) {
                while (sum > k){
                    sum -= sequence[start++];
                }
                
                if (sum == k) list.add(new int[] {start, end});
            }
            end++;
        }
        list.sort((a, b) -> {
            int arrA = a[1] - a[0];
            int arrB = b[1] - b[0];
            
            return arrA == arrB ? a[0] - b[0] : arrA - arrB;
        });
        
        return list.get(0);
    }
}
