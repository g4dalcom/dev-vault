class Solution {
    public int[] solution(int brown, int yellow) {
        int col = 3;
        int row = (brown+yellow) / col;

        while (row >= col) {
            if (row * col == brown + yellow && 2 * (row + col) - 4 == brown) break;

            col++;
            row = (brown+yellow) / col;
        }

        return new int[] {row, col};
    }
}
