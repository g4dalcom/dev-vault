import java.util.*;

class Solution {
    public int solution(int n, int m, int[] section) {
        Queue<Integer> q = new LinkedList<>();
        for (int i = 0; i < section.length; i++) {
            q.offer(section[i]);
        }
        
        int cnt = 0;
        while (!q.isEmpty()) {
            int tmp = q.peek();
            
            for (int i = tmp; i < tmp+m; i++) {
                if (q.contains(i)) {
                    q.poll();
                }
            }
            cnt++;
        }
        return cnt;
    }
}
